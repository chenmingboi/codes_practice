#### 高精度加法实现模板

```cpp
vector<int>add_gao(const vector<int>& a, const vector<int>& b) {
    int len1 = a.size(), len2 = b.size();
    int len = max(len1, len2);
    vector<int>result;
    result.reserve(len+1);
    int carry = 0;//进位
    for(int i = 0;i < len;i++) {
        int sum = carry;
        if(i < len1) sum += a[i];
        if(i < len2) sum += b[i];
        result.push_back(sum % 10);
        carry = sum / 10;
    }
    if(carry) result.push_back(carry);//最后一位有进位
    return result;
}
```

#### 高精度乘法实现模板

```c++
vector<int>mul_gao(const vector<int>&a, const vector<int>& b) {
    int len1 = a.size(), len2 = b.size();
    //乘法，结果位数最多是两个乘数位数之和
    int len = len1 + len2;
    vector<int>result(len, 0);
    for(int i = 0;i < len1;i++) {
        for(int j = 0;j < len2;j++) {
            result[i+j] += a[i] * b[j];
        }
    }
    for(int i = 0;i < len-1;i++) {
        result[i+1] += result[i] / 10;
        result[i] %= 10;
    }
    //去除前导0，并至少保留一位0
    while(result.size() > 1 && result.back() == 0) result.pop_back();
    return result;
}
```
